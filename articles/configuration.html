FigNet Application Tag

<?xml version="1.0" encoding="utf-16"?>
<Configuration
  <ApplicationType>Server</ApplicationType>		<!-- Server|Client -->
  <FrameRate>33</FrameRate>         <!-- Application tick rate per second -->
  <EnableFileLogs xsi:nil="true" /> <!-- optional -->
  <MTU xsi:nil="true" />            <!-- optional , you can remove that tag -->
  <MaxInBound xsi:nil="true" />     <!-- optional -->
  <MaxOutBound xsi:nil="true" />    <!-- optional -->
</Configuration>

ServerConfig Tag

Used to create IServerSocket

<ServerConfig>
    <Name>DefaultServer</Name>
    <Port>5555</Port>
    <Provider>ENet</Provider>                  <!-- ENet|LiteNetLib|TCP|WebSockets -->
    <MaxChannels xsi:nil="true" />             <!-- optional, can be removed if unused -->
    <MaxConnections xsi:nil="true" />          <!-- optional, can be removed if unused -->
</ServerConfig>

Modules Tag

Add under ServerConfig Tag

<Modules>                                     
    <Module>
      <AssemblyName>FigNetCore</AssemblyName>  <!-- AssemblyName -->
      <Type>FigNetCore.Rooms.Room</Type>       <!-- Class name with NameSpace -->
    </Module>
    <Module>
      <AssemblyName>AwesomeGame</AssemblyName>
      <Type>AwesomeGame.GameState</Type>
    </Module>
</Modules>

PeerConfigs Tag

Used to create IClientSocket, you can define multiple PeerConnections

<PeerConfigs>
  <PeerConfig>
    <Name>DefaultClient</Name>   
    <Port>5555</Port>
    <PeerIp>127.0.0.1</PeerIp>
    <Provider>ENet</Provider>
    <AutoConnect>true</AutoConnect>
    <MaxChannels xsi:nil="true" />
    <MaxReconnectTries xsi:nil="true" />
  </PeerConfig>
</PeerConfigs>
